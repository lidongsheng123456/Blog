```vue
<script>
  import {chat, getMessages} from "@/api/chat";
  import moment from 'moment'; // 用于格式化时间

  export default {
    data() {
      return {
        // 假设用户ID为1
        userId: 1,
        //用户输入信息
        userInput: '',
        //所有信息
        messages: []
      };
    },
    methods: {
      // async 函数：声明一个函数为异步函数，允许在函数内部使用 await 关键字。
      // async 函数总是返回一个 Promise
      async sendMessage() {
        // 如果去取字符串左右两边空字符还是空则退出方法
        if (this.userInput.trim() === '') return;
        let userInputMessage = this.userInput
        this.userInput = '';
        // 构造用户信息 类型为用户 内容为你加上用户的信息加上现在时间
        const userMessage = {type: 'user', content: 'You: ' + userInputMessage, createdAt: new Date()};
        // 将当前用户发送的信息推送到信息数据中
        this.messages.push(userMessage);

        // 可能会发生异常
        try {
          // await 表达式：用于等待一个 Promise 解析，并暂停当前函数的执行，直到 Promise 解析或拒绝。
          // await 只能在 async 函数中使用。
          // 注：await 只是暂停了 sendMessage 函数的执行，而不会阻塞主线程，意味着函数外面的代码会正常往下执行，而不会因为当前函数阻塞
          // 对象解构将chat返回的对象中的值提取出来（即函数的引用，当前函数允许逐块读取数据和处理流式数据）
          const {read} = await chat({userId: this.userId, message: userInputMessage});

          // 构造机器人回复的信息
          let currentMessage = {type: 'robot', content: 'Robot: ', createdAt: new Date()};
          // 推送
          this.messages.push(currentMessage);
          // 调用read方法每次调用时都会从响应体流中读取一块数据，并将其解码为文本返回
          let chunk = await read();
          // done返回true时代表已经读取完毕，false代表还有数据可以读取，取反继续执行
          while (!chunk.done) {
            currentMessage.content += chunk.value;
            // 更新信息数组中最后一个索引的数据
            this.$set(this.messages, this.messages.length - 1, currentMessage); // Update the last message
            // 继续调用read()将读取的结果返回即{done, value: chunk}
            chunk = await read();
          }
        } catch (error) {
          // 如果异常则往信息里面推送error类型信息
          this.messages.push({type: 'error', content: 'Error: ' + error.message, createdAt: new Date()});
        } finally {
          // 清空输入框
          this.userInput = '';
          this.scrollToBottom();
        }
      },
      // 将容器滚动到底部
      scrollToBottom() {
        // 使用 this.$nextTick 确保 DOM 已更新。
        this.$nextTick(() => {
          // 获取包含类名为 .messages 的元素
          const container = this.$el.querySelector('.messages');
          // 设置该元素的 scrollTop 属性为 scrollHeight，使其滚动到底部。
          container.scrollTop = container.scrollHeight;
        });
      },
      // 异步函数获取当前用户和机器人所有信息
      async fetchMessages() {
        try {
          const response = await getMessages(this.userId);
          // 迭代返回新数组对象
          this.messages = response.map(msg => ({
            type: msg.messageType,
            content: msg.content,
            createdAt: new Date(msg.createdAt)
          }));
        } catch (error) {
          console.error('Error fetching messages:', error);
        }
      },
      // 格式化时间
      formatTime(timestamp) {
        return moment(timestamp).format('YYYY-MM-DD HH:mm:ss');
      }
    },
    mounted() {
      // 获取当前用户所有信息，注册成功回调执行滚动到底部
      this.fetchMessages().then(() => {
        this.scrollToBottom();
      });
    }
  };
</script>
```

```javascript
// 2.fetch API
import request from '@/utils/request';

/**
 * 第二个 read 函数的执行次数取决于后端响应体流中的数据块数量。
 * 具体来说，read 函数每次调用时会从流中读取一块数据，
 * 直到流结束（即所有数据块都已读取完毕）。
 * @param userMessage
 * @returns {Promise<{read: function(): *}>}
 */
export function chat(userMessage) {

    //fetch APIs
    //在 chat 函数中使用 fetch 主要是为了处理流式数据，而 getMessages 函数使用 axios 则是因为它更适合处理简单的 GET 请求。
    return fetch('http://localhost:8081/chat', {
        method: 'POST',
        // 需要手动指定请求体为application/json
        headers: {
            'Content-Type': 'application/json'
        },
        // 将 JavaScript 对象 userMessage 转换为 JSON 格式的字符串。
        body: JSON.stringify(userMessage)
        // 第一个then处理网络响应的状态和初步解析
    }).then(response => {
        //注册成功回调
        // 如果响应状态码不在200-299范围内，抛出错误
        if (!response.ok) {
            throw new Error('网络响应不正常');
        }

        // 否则返回响应体
        return response.body;
        // 第二个then处理响应体(stream)
    }).then(body => {
        // 注册成功回调
        // 获取响应体中的读取器对象，用于逐块读取响应体流。
        const reader = body.getReader();
        // 创建一个 UTF-8 解码器，用于将二进制数据转换为文本。
        const decoder = new TextDecoder('utf-8');

        // read 函数用于从响应体流中读取数据并解码。
        function read() {
            // 调用响应体中读取器的read()方法读取流中的数据
            // read()返回Promise对象因为只有Promise对象才能通过.then()方法来处理异步操作的结果
            // 当响应体流中的数据可用时，Promise会解析为一个对象{done,value}
            // done:布尔值，当 done 为 true 时，表示所有数据都已经读取完毕；当 done 为 false 时，表示还有更多数据可以读取。
            // value:包含流中的数据块
            return reader.read().then(({done, value}) => {
                // 当流结束时表示没有更多的数据可以从流中读取了，将value设置为空字符串表示没有实际的数据
                if (done) {
                    console.log('Stream complete');
                    return {done, value: ''};
                }
                // 当流没结束时调用解码器将value解码为一个数据块，并设置流模式
                const chunk = decoder.decode(value, {stream: true});
                // 打印流中的数据块
                console.log('Received chunk:', chunk);
                // 返回 done:布尔值，是否完成;chunk:解码后的数据块
                return {done, value: chunk};
            });
        }

        // 返回一个对象该对象中包含一个名为read的属性，值为read函数的引用,允许逐块读取和处理流式数据。
        return {read};
    });
}

export function getMessages(userId) {
    return request.get(`/chat/messages/${userId}`);
}
```

```java
// 1.控制层
package com.clh.controller;

import com.alibaba.dashscope.app.Application;
import com.alibaba.dashscope.app.ApplicationParam;
import com.alibaba.dashscope.app.ApplicationResult;
import com.alibaba.dashscope.exception.ApiException;
import com.alibaba.dashscope.exception.InputRequiredException;
import com.alibaba.dashscope.exception.NoApiKeyException;
import com.clh.pojo.ChatMessage;
import com.clh.service.ChatMessageService;
import io.reactivex.Flowable;
import lombok.Getter;
import lombok.Setter;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.servlet.mvc.method.annotation.StreamingResponseBody;

import java.io.IOException;
import java.util.List;

@Slf4j
@RestController
public class ChatController {

    @Autowired
    private ChatMessageService chatMessageService;

    /**
     * 处理聊天请求，实现与通义-Max的流式交互。
     *
     * @param request 包含用户ID和消息内容的聊天请求对象
     * @return 返回一个流式响应，包含通义-Max的回复
     */
    @PostMapping(value = "/chat", produces = MediaType.APPLICATION_OCTET_STREAM_VALUE)
    public ResponseEntity<StreamingResponseBody> chat(@RequestBody ChatRequest request) {
        // 记录日志信息
        log.info("调用通义-Max,{}", request);

        // 获取用户ID和消息内容
        int userId = request.getUserId();
        String userMessage = request.getMessage();

        // 保存用户消息到数据库
        ChatMessage userChatMessage = new ChatMessage();
        userChatMessage.setUserId(userId);
        userChatMessage.setMessageType("user");
        userChatMessage.setContent("You: " + userMessage);
        chatMessageService.saveMessage(userChatMessage);

        // 返回流式响应
        return ResponseEntity.ok()
                .contentType(MediaType.APPLICATION_OCTET_STREAM)
                .body(outputStream -> {
                    try {
                        // 构建应用参数
                        ApplicationParam param = ApplicationParam.builder()
                                .apiKey(System.getenv("DASHSCOPE_API_KEY"))
                                .appId("268c83f42290484b81019c4cb7ee53dc")
                                .prompt(userMessage)
                                .incrementalOutput(true)
                                .build();

                        // 创建应用实例并发起流式调用
                        Application application = new Application();
                        Flowable<ApplicationResult> result = application.streamCall(param);

                        // 初始化机器人消息对象
                        ChatMessage robotChatMessage = new ChatMessage();
                        robotChatMessage.setUserId(userId);
                        robotChatMessage.setMessageType("robot");
                        robotChatMessage.setContent("Robot: ");
                        chatMessageService.saveMessage(robotChatMessage);

                        // 处理流式返回的数据
                        result.blockingForEach(data -> {
                            // 获取每次返回的文本片段
                            String segment = data.getOutput().getText();
                            // 更新机器人消息内容并保存到数据库
                            robotChatMessage.setContent(robotChatMessage.getContent() + segment);
                            chatMessageService.saveMessage(robotChatMessage);
                            // 将文本片段写入输出流并刷新
                            outputStream.write((segment + "\n").getBytes());
                            outputStream.flush();
                        });
                    } catch (ApiException | NoApiKeyException | InputRequiredException e) {
                        // 捕获异常并记录错误消息
                        ChatMessage errorChatMessage = new ChatMessage();
                        errorChatMessage.setUserId(userId);
                        errorChatMessage.setMessageType("error");
                        errorChatMessage.setContent("Exception: " + e.getMessage());
                        chatMessageService.saveMessage(errorChatMessage);
                        // 将异常信息写入输出流并刷新
                        outputStream.write(("Exception: " + e.getMessage() + "\n").getBytes());
                        outputStream.flush();
                    } finally {
                        // 关闭输出流
                        try {
                            outputStream.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                });
    }


    @GetMapping("/chat/messages/{userId}")
    public List<ChatMessage> getMessages(@PathVariable int userId) {
        log.info("查看机器人和你的聊天记录,{}", userId);
        return chatMessageService.getMessagesByUserId(userId);
    }
}

@Setter
@Getter
class ChatRequest {
    private int userId;
    private String message;
}
```

### 这聊天实现也涉及我没接触过的技术，在这里说明一下

* dashscope-sdk-java
    1. 这是阿里云提供的一个Java SDK，用于简化与阿里云服务的交互。它封装了对阿里云API的调用，使得开发者可以通过简单的函数调用来使用阿里云的服务，如模型训练、推理等。
* fetch
    1. fetch 是一种用于发起网络请求的API，通常在前端JavaScript中使用。它可以替代传统的XMLHttpRequest，提供更简洁的语法和更强大的功能，比如支持Promise，可以更容易地处理异步操作。
* StreamingResponseBody
    1. 这是Spring框架提供的一个接口，用于实现流式响应。
    2. 当需要向客户端发送大量数据时，可以使用StreamingResponseBody来逐块写入响应体，而不是一次性将所有数据加载到内存中再返回给客户端。这种方式对于处理大文件下载、实时日志流等场景非常有用。
* Spring Data JPA
    1. Spring Data JPA是Spring框架的一部分，旨在简化数据访问层的开发。它基于JPA（Java Persistence
       API）构建，提供了声明式的CRUD操作以及查询方法的支持，极大地减少了编写重复性数据访问代码的工作量。通过定义接口并遵循一定的命名规则，就可以轻松实现复杂的数据库查询逻辑。
    2. 实体管理器自动处理
        * 当你调用 save 方法时，Spring Data JPA 会根据传入的实体类的状态自动执行相应的数据库操作：
        * 如果实体是新创建的（即没有主键或主键为默认值），则会执行插入操作。
        * 如果实体已经存在（即有有效的主键），则会执行更新操作。
    3. 事务管理
        * 通常，save 方法会在一个事务中执行。Spring 会确保整个操作要么全部成功，要么全部回滚，保证数据的一致性。
    4. 你不需要显式编写 SQL 语句的原因是 JPA 框架已经帮你封装好了这些底层操作。你可以专注于业务逻辑，而由框架来处理与数据库的交互。
    5. 约定大于配置重要！！！
